program fibonacci;

int res;

func int fib (int n) {
    int temp1; 
    int temp2;

    if (n == 0) {
        return(0);
    } else {
        if (n == 1) {
            return(1);
        } else {
            temp1 = fib(n - 1);
            temp2 = fib(n - 2);
            return (temp1 + temp2);
        }
    }
}

main() {
    res = fib(5);
    print(res);
};
